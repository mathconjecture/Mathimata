\section*{Όρια}

\begin{enumerate}

  \item Να υπολογιστούν τα παρακάτω όρια:

    \twocolumnside{
      \begin{enumerate}[(i)]
        \item $\lim_{(x,y)\to (0,0)}\frac{ x^{2}-y^{2} }{ \sqrt{x}-\sqrt{ y } } $ 
          \hfill Απ: $0 $ 
        \item $ \lim\limits_{(x,y)\to (0, 0)} \frac{\sqrt{x^{2}+y^{2}+4} - 2}
          {x^{2}+y^{2}} $ \hfill Απ: $ \frac{1}{4} $ 
      \end{enumerate}}{
      \begin{enumerate}[(i),start=3]
      % \item $ \lim_{(x,y)\to (0,0)} \frac{ x^{4} + y^{4} }{ x^{2} + y^{2} } $
      %   \hfill Απ: $0 $ 
      \item $ \lim_{(x,y)\to (0,0)} \frac{ x^{2}y }{ x^{2} + y^{2} } $ \hfill Απ: $0 $ 
      \item $ \lim_{(x,y)\to (0,0)} \frac{ x^{2}y^{3} }{ x^{2} + y^{2} } $\hfill Απ: $0$ 
        % \item $ \lim_{(x,y)\to (0,0)} \frac{ x^{2}y^{3} }{ x^{2} +y^{4} } $
        % \hfill Απ: $0 $ 
  \end{enumerate}}

  \item Να δείξετε ότι τα παρακάτω όρια δεν υπάρχουν

    \twocolumnside{
      \begin{enumerate}[(i)]
        \item $ \lim_{(x,y)\to (0,0)} \frac{x^{2} - y^{2} }{ x^{2} + y^{2} } $
        \item $ \lim_{(x,y)\to (0,0)} \frac{x + y}{x - 2y} $ 
        \item $ \lim_{(x,y)\to (0,0)} \frac{x^{4} + y }{ x + y^{4}} $ 
      \end{enumerate}}{
      \begin{enumerate}[(i),start=4]
        \item $ \lim_{(x,y)\to (0,0)} \frac{2x^{2}y }{ x^{4} + y^{2}  } $
        \item $ \lim_{(x,y)\to (0,0)} \frac{xy^{3}}{x^{2}+y^{6}} $
        \item $ \lim_{(x,y)\to (0,0)} \frac{xy^{3} + 2y^{4} }{ x^{4}+y^{4}}  $
          % \item $ \lim_{(x,y)\to (-2,5)} \frac{2x+y-1}{x-y+7} $
    \end{enumerate}}
\end{enumerate}



\section*{Συνέχεια}

\begin{enumerate}
  \item Να εξεταστούν πλήρως ως προς τη συνέχεια οι συνάρτησεις:
    \begin{enumerate}[(i)]
      \item $ f(x,y) = 
        \begin{cases}
          \frac{ xy }{ x^{2} + y^{2} } ,& (x,y)\neq (0,0) \\
          0 ,& (x,y)= (0,0)
        \end{cases} $
        \hfill Απ: ασυνεχής στο $ (0,0) $ 

      \item $ f(x,y) = 
        \begin{cases} 
          \frac{y^{2}}{x^{2}+y^{2}}, & (x,y) 
          \neq (0,0) \\ 0, & (x,y) = (0,0)
        \end{cases} $
        \hfill Απ: ασυνεχής στο $ (0,0) $ 
    \end{enumerate}
\end{enumerate}
