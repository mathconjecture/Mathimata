\begin{enumerate}
  \item Έστω οι αλγεβρικές δομές $ (\mathbb{N},+,\cdot), (\mathbb{Z},+,\cdot),
    (\mathbb{Q},+,\cdot), (\mathbb{R},+,\cdot) $ και $ (\mathbb{C},+,\cdot) $, 
    όπου οι πράξεις $ "+" $ και $ "\cdot" $ είναι η συνήθης πρόσθεση και πολλαπλασιασμός.
    \begin{myitemize}
      \item Οι πράξεις είναι προσεταιριστικές και αντιμεταθετικές.
      \item Ο πολλαπλασιασμός είναι επιμεριστικός ως προς την πρόσθεση.
      \item Το ουδέτερο στοιχείο της πρόσθεσης είναι το $ 0 $ και το ουδέτερο στοιχείο 
        του πολλαπλασιασμού είναι το $ 1 $.
      \item Κάθε μη μηδενικό στοιχείο του $ \mathbb{N} $ δεν έχει συμμετρικό ως προς την
        πρόσθεση.
      \item Κάθε στοιχείο του $ \mathbb{N} - \{ 1 \} $ δεν έχει συμμετρικό ως προς τον 
        πολλαπλασιασμό.
      \item Κάθε στοιχείο $ x \in \mathbb{Z} $ έχει συμμετρικό ως προς την πρόσθεση, 
        το $ -x \in \mathbb{Z} $, ενώ κανένα στοιχείο του $ \mathbb{Z} - \{ -1,1 \} $ 
        δεν έχει συμμετρικό ως προς τον πολλαπλασιασμό.
      \item Κάθε στοιχείο $ x \in \mathbb{Q}, \mathbb{R}, \mathbb{C} $ έχει συμμετρικό
        ως προς την πρόσθεση, το $ -x $ και κάθε μη μηδενικό στοιχείο $ x \in \mathbb{Q},
        \mathbb{R}, \mathbb{C}$ έχει συμμετρικό το $ \frac{1}{x} $.
    \end{myitemize}
\end{enumerate}

\subsection*{Ομάδες}

\twocolumnsides{
  \begin{myitemize}
    \item $ (\mathbb{N}, +) $ δεν είναι ομάδα
    \item $ (\mathbb{Z},+) $ αβελιανή ομάδα 
    \item $ (\mathbb{Q},+) $ αβελιανή ομάδα 
    \item $ (\mathbb{R},+) $ αβελιανή ομάδα 
    \item $ (\mathbb{C}, +) $ αβελιανή ομάδα 
  \end{myitemize}
}{
  \begin{myitemize}
    \item $ (\mathbb{N} - \{ 0 \},\cdot) $ δεν είναι ομάδα
    \item $ (\mathbb{Z} - \{ 0 \} ,\cdot) $ δεν είναι ομάδα
    \item $ (\mathbb{Q} - \{ 0 \} , \cdot) $ αβελιανή ομάδα 
    \item $ (\mathbb{R} - \{ 0 \} , \cdot) $ αβελιανή ομάδα 
    \item $ (\mathbb{C} - \{ 0 \}, \cdot ) $ αβελιανή ομάδα 
  \end{myitemize}
}


\subsection*{Δακτύλιοι}

\begin{myitemize}
  \item $ (\mathbb{Z}, +,\cdot) $ αντιμεταθετικός δακτύλιος με μοναδιαίο στοιχείο το 1 
  \item $ (\mathbb{Q}, +,\cdot) $ αντιμεταθετικός δακτύλιος με μοναδιαίο στοιχείο το 1 
  \item $ (\mathbb{R}, +,\cdot) $ αντιμεταθετικός δακτύλιος με μοναδιαίο στοιχείο το 1 
  \item $ (\mathbb{C}, +,\cdot) $ αντιμεταθετικός δακτύλιος με μοναδιαίο στοιχείο το 1 
\end{myitemize}
